  1080→interface I18nContextValue {
  1081→  locale: Locale
  1082→  setLocale: (locale: Locale) => void
  1083→  t: (key: string, params?: Record<string, string>) => string
  1084→}
  1085→
  1086→const I18nContext = createContext<I18nContextValue | null>(null)
  1087→
  1088→export function I18nProvider({ children }: { children: ReactNode }) {
  1089→  // Initialize from localStorage / browser language synchronously (SSR-safe with fallback)
  1090→  const [locale, setLocaleState] = useState<Locale>(() => {
  1091→    if (typeof window === "undefined") return "it"
  1092→    try {
  1093→      const saved = localStorage.getItem("aperture-locale") as Locale | null
  1094→      if (saved && LOCALES.some((l) => l.code === saved)) return saved
  1095→      const browserLang = navigator.language.toLowerCase()
  1096→      const matched = LOCALES.find((l) => browserLang.startsWith(l.code))
  1097→      return matched ? matched.code : "it"
  1098→    } catch {
  1099→      return "it"
  1100→    }
  1101→  })
  1102→
  1103→  const setLocale = (newLocale: Locale) => {
  1104→    setLocaleState(newLocale)
  1105→    try {
  1106→      localStorage.setItem("aperture-locale", newLocale)
  1107→    } catch {}
  1108→  }
  1109→
  1110→  const t = (key: string, params?: Record<string, string>): string => {
  1111→    let value = messages[locale]?.[key] ?? messages.en[key] ?? key
  1112→    if (params) {
  1113→      for (const [k, v] of Object.entries(params)) {
  1114→        value = value.replace(`{${k}}`, v)
  1115→      }
  1116→    }
  1117→    return value
  1118→  }
  1119→
  1120→  return (
  1121→    <I18nContext.Provider value={{ locale, setLocale, t }}>
  1122→      {children}
  1123→    </I18nContext.Provider>
  1124→  )
  1125→}
  1126→
  1127→export function useI18n() {
  1128→  const ctx = useContext(I18nContext)
  1129→  if (!ctx) {
  1130→    throw new Error("useI18n must be used within I18nProvider")
  1131→  }
  1132→  return ctx
  1133→}
  1134→
  1135→// Convenience hook for just the translation function
  1136→export function useT() {
  1137→  return useI18n().t
  1138→}
